<?php
$message = '';
$error = '';

if (isset($_POST['create'])) {
    $dir = rtrim($_POST['dir'], '/');
    $filename = basename($_POST['filename']);
    $content = $_POST['content'];

    if (empty($dir) || empty($filename)) {
        $error = 'Directory and filename are required.';
    } elseif (!preg_match('/^[\w\-. ]+$/', $filename)) {
        $error = 'Invalid filename. Use only letters, numbers, dashes, dots.';
    } else {
        if (!is_dir($dir)) {
            mkdir($dir, 0755, true);
        }
        $fullPath = $dir . '/' . $filename;
        if (file_put_contents($fullPath, $content) !== false) {
            $message = "File created: " . htmlspecialchars($fullPath);
        } else {
            $error = 'Failed to write file. Check permissions.';
        }
    }
}

// Resolve browse path - allow going all the way to root
$requestedPath = $_GET['path'] ?? null;
if ($requestedPath !== null) {
    $browse = realpath($requestedPath);
    if (!$browse || !is_dir($browse)) $browse = '/';
} else {
    $browse = realpath('.');
}

// Build breadcrumb parts from path
function getBreadcrumbs($path) {
    $parts = explode('/', trim($path, '/'));
    $crumbs = [['label' => '/', 'path' => '/']];
    $current = '';
    foreach ($parts as $part) {
        if ($part === '') continue;
        $current .= '/' . $part;
        $crumbs[] = ['label' => $part, 'path' => $current];
    }
    return $crumbs;
}

            echo " <a href='?path=" . urlencode($parent) . "'>.. up to " . htmlspecialchars($parent) . "</a>";

        $items = @scandir($browse);
        if ($items) {
            // Dirs first, then files
            $dirs = $files = [];
            foreach ($items as $item) {
                if ($item === '.' || $item === '..') continue;
                $full = $browse . '/' . $item;
                if (is_dir($full)) $dirs[] = $item;
                else $files[] = $item;
            }
            sort($dirs); sort($files);

            foreach ($dirs as $item) {
                $full = $browse . '/' . $item;
                echo "<div class='tree-item'>";
                echo "f <a href='?path=" . urlencode($full) . "&selectdir=" . urlencode($full) . "'>" . htmlspecialchars($item) . "</a>";
                echo "</div>";
            }
            foreach ($files as $item) {
                $full = $browse . '/' . $item;
                $size = filesize($full);
                $size = $size > 1048576 ? round($size/1048576,1).'MB' : ($size > 1024 ? round($size/1024,1).'KB' : $size.'B');
                echo "<div class='tree-item'>";
                echo "F <span style='flex:1'>" . htmlspecialchars($item) . "</span>";
                echo "<span class='size'>{$size}</span>";
                echo "</div>";
            }
        } else {
            echo "<div style='padding:14px;color:#999;font-size:12px'>No access or empty.</div>";
